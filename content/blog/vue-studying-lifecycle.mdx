## Lifecycle Methods

- *onBeforeMount* - called before mountinf begins.
- *onMounted* - called when component in mounted.
- *onBeforeUpdate* - Called when reactive data changes and before re-render.
- *onUpdate* - Called after re-render.
- *onBeforeUnmount* - called before the Vue instalnce is destroyed.
- *onUnmounted* - Called after the instance is destroyed.
- *onActivated* - called when a kep-alive component is activated.
- *onDeactivated* - called when a kept-alive component is deactivated.
- *onErrorCaptured* - called when an erro is caputure fromm a child component.

        <div v-for="(value, propiedad, index) in frutas" :key="index">

```js {2} showLineNumbers
<script setup>
import { ref, computed } from 'vue';
const name = 'Dinamic See';

const counter = ref(0);
const arrayFavorite = ref([]);

const increment = () => counter.value++;
const decrement = () => {
  if (counter.value > 0) {
    counter.value--;
  }
};
const reset = () => (counter.value = 0);
const add = () => {
  arrayFavorite.value.push(counter.value);
};
const disabledBtn = computed(() => {
  const numSearch = arrayFavorite.value.find((num) => num === counter.value);
  return numSearch || numSearch === 0;
});
</script>

<template>
  <div class="container">
    <h1>
      Hi, {{ name.toUpperCase() }} -
      <span :class="[counter <= 0 ? 'red' : 'green']">{{ counter }}</span>
    </h1>
    <div class="d-flex flex-row justify-content-center gap-3">
    <button @click="increment" class="btn btn-outline-success">+</button>
    <button @click="decrement" class="btn btn-outline-danger" :disabled="counter <= 0">-</button>
    <button @click="reset">Reset</button>
    <button @click="add" :disabled="disabledBtn">add</button>
    </div>
    <br />
    {{ arrayFavorite }}
  </div>
</template>

<style scoped>
ul {
  text-align: left;
}
.red {
  color: red;
}
.green {
  color: green;
}
</style>

```
